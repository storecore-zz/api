  5  * StoreCore Database Backup
  6  *

 10  * @package   StoreCore\Core
 11  * @version   0.1.0
 12  *
 13  * @todo
 14  *   Because the `SHOW TABLES` command lists database tables in alphabetical
 15  *   order, the recovery of a full backup may fail on chained `FOREIGN KEY`
 16  *   constraints.  Depending on the database structure and the selected
 17  *   tables, you MAY have to manually change the order of the `CREATE TABLE`
 18  *   statements.
 19  */
 20 class Backup
 21 {
 22     /** @var string VERSION Semantic Version (SemVer) */
 23     const VERSION = '0.1.0';
 24 
 25     /**
 26      * Save a database backup to file.
 27      *
 28      * @param string|array $tables
 29      *   Optional comma-separated string or array with the names of database
 30      *   tables to save.  Defaults to '*' for all tables.
 31      *
 32      * @param bool $drop_tables
 33      *   If set to true, `DROP TABLE <tbl_name>` clauses are added to delete
 34      *   existing database tables.  Defaults to false, so only non-existent
 35      *   database tables are recreated with the `CREATE TABLE IF NOT EXISTS`
 36      *   clause.
 37      *
 38      * @return string|false
 39      *   Returns the filename of the SQL backup file on success or false on
 40      *   failure.
 41      */
 42     public static function save($tables = '*', $drop_tables = false)
 43     {
 44         // Get or create a logger.
 45         $registry = \StoreCore\Registry::getInstance();
 46         if ($registry->has('Logger')) {
 47             $logger = $registry->get('Logger');
 48         } elseif (STORECORE_NULL_LOGGER) {
 49             $logger = new \Psr\Log\NullLogger();
 50         } else {
 51             $logger = new \StoreCore\FileSystem\Logger();
 52         }
 53 
 54         // Connect to the database with a new client.
 55         try {
 56             $dbh = new \StoreCore\Database\Connection();
 57         } catch (\PDOException $e) {
 58             $logger->error('Database connection error ' . $e->getCode() . ': ' . $e->getMessage());
 59             return false;
 60         }
 61 
 62         // Fetch (all) tables
 63         if ($tables == '*') {
 64             try {
 65                 $result = $dbh->query('SHOW TABLES');
 66                 if ($result === false) {
 67                     return false;
 68                 } else {
 69                     $tables = array();
 70                     foreach ($result as $row) {
 71                         $tables[] = $row[0];
 72                     }
 73                 }
 74                 $result->closeCursor();
 75             } catch (\PDOException $e) {
 76                 $logger->error('Database error ' . $e->getCode() . ' showing tables: ' . $e->getMessage());
 77                 return false;
 78             }
 79         } else {
 80             $tables = is_array($tables) ? $tables : explode(',', $tables);
 81         }
 82 
 83         // Write pending changes to file.
 84         try {
 85             $dbh->exec('FLUSH TABLES');
 86         } catch (\PDOException $e) {
 87             $logger->notice('Error ' . $e->getCode() . ' flushing tables: ' . $e->getMessage());
 88         }
 89 
 90         // Cycle through all tables
 91         try {
 92             $sql_file_contents = (string)null;
 93             foreach ($tables as $table_name) {
 94                 $sql = (string)null;
 95                 $prepend = false;
 96 
 97                 // Add an optional DROP TABLE clause.
 98                 if ($drop_tables) {
 99                     $sql .= 'DROP TABLE ' . $table_name . ";\n\n";
100                 }
101 
102                 // Add the CREATE TABLE IF NOT EXISTS clause.
103                 $result = $dbh->query('SHOW CREATE TABLE ' . $table_name);
104                 $create_table_row = $result->fetch(\PDO::FETCH_NUM);
105                 $create_table = preg_replace('/CREATE TABLE/', 'CREATE TABLE IF NOT EXISTS', $create_table_row[1], 1);
106                 $sql .= $create_table . ";\n\n";
107 
108                 // Tables without foreign keys are prepended.
109                 if (strpos($create_table, ' FOREIGN KEY ') === false) {
110                     $prepend = true;
111                 }
112 
113                 // Add the table data from a non-empty table.
114                 $result = $dbh->query('SELECT COUNT(*) FROM ' . $table_name);
115                 if ($result->fetchColumn() > 0) {
116                     $result = $dbh->query('SELECT * FROM ' . $table_name);
117                     $sql .= 'INSERT IGNORE INTO ' . $table_name . ' VALUES';
118                     while ($row = $result->fetch(\PDO::FETCH_NUM)) {
119                         foreach ($row as $key => $value) {
120                             if (is_string($value)) {
121                                 $row[$key] = "'" . addslashes($value) . "'";
122                             }
123                         }
124                         $sql .= "\n  (" . implode(',', $row) . '),';
125                     }
126                     $sql = rtrim($sql, ',') . ";\n\n";
127                 }
128 
129                 // Prepend or append
130                 if ($prepend) {
131                     $sql_file_contents = $sql . $sql_file_contents;
132                 } else {
133                     $sql_file_contents = $sql_file_contents . $sql;
134                 }
135             }
136         } catch (\PDOException $e) {
137             $logger->error('Database error ' . $e->getCode() . ': ' . $e->getMessage());
138             return false;
139         }
140 
141         // Disconnect
142         try {
143             $result->closeCursor();
144             $dbh = null;
145             unset($result, $dbh);
146         } catch (\PDOException $e) {
147             $logger->notice('Error ' . $e->getCode() . ' flushing tables: ' . $e->getMessage());
148         }
149 
150         // Save SQL file
151         if (defined('STORECORE_FILESYSTEM_CACHE_DIR')) {
152             $dir = STORECORE_FILESYSTEM_CACHE_DIR;
153         } else {
154             $dir = __DIR__ . DIRECTORY_SEPARATOR;
155         }
156         $filename = $dir . 'backup-' . gmdate('Y-m-d-H-m-s') . '-UTC.sql';
157         $handle = fopen($filename, 'w+');
158         if ($handle !== false) {
159             fwrite($handle, $sql_file_contents);
160             fclose($handle);
161             $logger->notice('Database backup saved as ' . $filename . '.');
162             return $filename;
163         } else {
164             $logger->error('Could not create database backup file.');
165             return false;
166         }
167     }
168 }
